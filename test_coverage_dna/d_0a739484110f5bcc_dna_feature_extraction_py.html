9def extract_dna_features(folder_path: str) -> None: 

10 """Extract and compute DNA features for each CSV file containing genomic sequences. 

11 

12 Add new columns to the existing CSV files with computed codon frequencies, lengths of 

13 different DNA segments (utr5, cds, utr3), GC content in DNA segments, and GC content at 

14 wobble positions. 

15 

16 Args: 

17 folder_path (str): The path to the folder containing genomic data CSV files. 

18 

19 Returns: 

20 None: This function does not return a value but modifies the files in the specified directory. 

21 """ 

22 # Iterate over files in the directory 

23 for filename in os.listdir(folder_path): 

24 if filename.endswith(".csv"): 

25 file_path = os.path.join(folder_path, filename) 

26 print("Extracting DNA features from:", filename) 

27 

28 # Create a temporary file to write the modified data 

29 temp_file = tempfile.NamedTemporaryFile( 

30 mode="w", delete=False, newline="", encoding="utf-8" 

31 ) 

32 

33 # Open the CSV file for reading 

34 with open( 

35 file_path, "r", newline="", encoding="utf-8" 

36 ) as infile, temp_file: 

37 reader = csv.DictReader(infile) 

38 

39 # Define the fieldnames for the output CSV 

40 header = reader.fieldnames 

41 if header != [ 

42 "ensembl_gene_id", 

43 "transcript_id", 

44 "promoter", 

45 "utr5", 

46 "cds", 

47 "utr3", 

48 "terminator", 

49 ]: 

50 break 

51 new_columns = [] 

52 codons = [ 

53 "".join(combination) for combination in product("ACGT", repeat=3) 

54 ] 

55 new_columns.extend(codons) 

56 new_columns.extend( 

57 [ 

58 "cds_length", 

59 "utr5_length", 

60 "utr3_length", 

61 "utr5_gc", 

62 "cds_gc", 

63 "utr3_gc", 

64 "cds_wobble2_gc", 

65 "cds_wobble3_gc", 

66 ] 

67 ) 

68 header.extend(new_columns) 

69 

70 # Open the CSV file for writing 

71 writer = csv.DictWriter(temp_file, fieldnames=header) 

72 writer.writeheader() 

73 

74 # Compute features for each gene 

75 for row in reader: 

76 # Add data for new columns (assuming new_columns is a list of values) 

77 utr5 = row.get("utr5") 

78 cds = row.get("cds") 

79 utr3 = row.get("utr3") 

80 

81 row.update(compute_cds_codon_frequencies(cds=cds, codons=codons)) 

82 row.update(compute_lengths(cds=cds, utr5=utr5, utr3=utr3)) 

83 row.update( 

84 compute_gc_content_sequence_components( 

85 utr5=utr5, cds=cds, utr3=utr3 

86 ) 

87 ) 

88 row.update(compute_gc_content_wobble_positions(cds)) 

89 

90 # Write the modified row to the temporary file 

91 writer.writerow(row) 

92 

93 # Replace the original file with the temporary file 

94 shutil.move(temp_file.name, file_path)

97def compute_cds_codon_frequencies(cds: str, codons: List[str]) -> Dict[str, float]: 

98 """Compute the frequency of every possible codon in the cds. 

99 

100 Args: 

101 cds (str): cds sequence 

102 

103 codons (List[str]): list of strings representing all 64 possible codons 

104 e.g. "AGA" 

105 

106 Returns: 

107 Dict[str, float]: dictionary of codon frequencies for all 64 possible codons, 

108 with codons as keys and their frequencies as values. 

109 """ 

110 cds_length = len(cds) 

111 codon_count = cds_length // 3 

112 # Placeholder for codon frequencies 

113 codon_frequencies = {codon: 0 for codon in codons} 

114 

115 # Address situation in which the cds provided is too short 

116 if codon_count == 0: 

117 return codon_frequencies 

118 

119 # Compute codon frequencies based on the sequence in the row 

120 for i in range(0, cds_length - 2, 3): 

121 codon = cds[i : i + 3] 

122 if len(codon) == 3: 

123 codon_frequencies[codon] += 1 

124 

125 for codon in codon_frequencies: 

126 codon_frequencies[codon] /= codon_count 

127 

128 return codon_frequencies

131def compute_lengths(cds: str, utr5: str, utr3: str) -> Dict[str, int]: 

132 """Compute the length of the cds, utr3 and utr5 DNA sequences. 

133 

134 Args: 

135 cds (str): CDS DNA sequence. 

136 utr5 (str): 5' UTR DNA sequence. 

137 utr3 (str): 3' UTR DNA sequence. 

138 

139 Returns: 

140 Dict[str, int]: A dictionary with keys 'cds', 'utr5', and 'utr3', and integer values 

141 representing the lengths of these sequences. 

142 """ 

143 lengths = { 

144 "cds_length": len(cds) if len(cds) > 0 else "", 

145 "utr5_length": len(utr5) if len(utr5) > 0 else "", 

146 "utr3_length": len(utr3) if len(utr3) > 0 else "", 

147 } 

148 

149 return lengths

152def compute_gc_content_sequence_components( 

153 utr5: str, cds: str, utr3: str 

154) -> Dict[str, float]: 

155 """Compute the GC content of the cds, utr3 and utr5. 

156 

157 GC content is defined as (G + C)/(A + T + G + C), where each letter 

158 represents the number of times that nucleotide appears in the DNA sequence. 

159 

160 Args: 

161 utr5 (str): 5' UTR DNA sequence. 

162 cds (str): CDS DNA sequence. 

163 utr3 (str): 3'UTR DNA sequence. 

164 

165 Returns: 

166 Dict[str, float]: A dictionary with keys 'cds_gc', 'utr5_gc', and 'utr3_gc', 

167 and float values representing the GC content of these sequences. 

168 """ 

169 utr5_gc_count = count_gc_nucleotides(utr5) 

170 cds_gc_count = count_gc_nucleotides(cds) 

171 utr3_gc_count = count_gc_nucleotides(utr3) 

172 

173 utr5_length = len(utr5) 

174 cds_length = len(cds) 

175 utr3_length = len(utr3) 

176 

177 utr5_gc = utr5_gc_count / utr5_length if utr5_length != 0 else "" 

178 cds_gc = cds_gc_count / cds_length if cds_length != 0 else "" 

179 utr3_gc = utr3_gc_count / utr3_length if utr3_length != 0 else "" 

180 

181 gc_content = {"utr5_gc": utr5_gc, "cds_gc": cds_gc, "utr3_gc": utr3_gc} 

182 

183 return gc_content

186def count_gc_nucleotides(sequence: str) -> int: 

187 """Count the number of times G and C nucleotides appear in a DNA sequence. 

188 

189 Args: 

190 sequence (str): DNA sequence 

191 

192 Returns: 

193 int: Sum of G and C counts in the sequence. 

194 """ 

195 g_content = sequence.count("G") 

196 c_content = sequence.count("C") 

197 return g_content + c_content

200def compute_gc_content_wobble_positions(cds: str) -> Dict[str, float]: 

201 """Compute the GC content of wobble positions 2 and 3 in the cds. 

202 

203 GC content is defined as (G + C)/(A + T + G + C), where each letter 

204 represents the number of times that nucleotide appears in a given 

205 wobble position 

206 

207 Args: 

208 cds (str): CDS DNA sequence. 

209 

210 Returns: 

211 Dict[str, float]: A dictionary with keys 'wobble2_gc' and 'wobble3_gc', 

212 and float values representing the GC content at wobble positions 2 and 3 

213 in the cds, respectively. 

214 """ 

215 # Extract the nucleotides in positions 2 and 3 into separate lists 

216 wobble2_nucleotides = cds[1::3] 

217 wobble3_nucleotides = cds[2::3] 

218 

219 wobble2_gc_count = wobble2_nucleotides.count("G") + wobble2_nucleotides.count("C") 

220 wobble3_gc_count = wobble3_nucleotides.count("G") + wobble3_nucleotides.count("C") 

221 

222 # Compute the GC content 

223 wobble2_gc = ( 

224 wobble2_gc_count / len(wobble2_nucleotides) 

225 if len(wobble2_nucleotides) != 0 

226 else "" 

227 ) 

228 

229 wobble3_gc = ( 

230 wobble3_gc_count / len(wobble3_nucleotides) 

231 if len(wobble3_nucleotides) != 0 

232 else "" 

233 ) 

234 

235 gc_content = {"cds_wobble2_gc": wobble2_gc, "cds_wobble3_gc": wobble3_gc} 

236 

237 return gc_content
